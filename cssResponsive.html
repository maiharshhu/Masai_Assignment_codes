/* CSS Units Explained */ /* Absolute Units */ px /* Pixels - Fixed size */ pt /* Points - Print ke liye */ cm /* Centimeters */ /* Relative Units */ rem /* Root element (html) ke font-size ke relative */ em /* Parent element ke font-size ke relative */ % /* Parent element ke size ke relative */ /* Viewport Units */ vw /* Viewport width ka percentage (1vw = 1% of viewport width) */ vh /* Viewport height ka percentage (1vh = 1% of viewport height) */ vmin /* vw aur vh mein se jo chota ho */ vmax /* vw aur vh mein se jo bada ho */ /* Examples */ .responsive-text { font-size: clamp(1rem, 2.5vw, 3rem); /* Min, Preferred, Max */ }

6. Media Queries

Responsive Breakpoints
Screen size change karo aur dekho!
Yeh text responsive hai - screen size ke saath change hota hai
Responsive Image/Element
/* Mobile First Approach (Recommended) */ /* Base styles for mobile */ .container { width: 100%; padding: 1rem; } /* Tablet styles */ @media (min-width: 768px) { .container { max-width: 750px; padding: 2rem; } } /* Desktop styles */ @media (min-width: 1024px) { .container { max-width: 1200px; padding: 3rem; } } /* Common Breakpoints */ /* Extra small devices (phones, 600px and down) */ @media only screen and (max-width: 600px) { } /* Small devices (portrait tablets and large phones, 600px and up) */ @media only screen and (min-width: 600px) { } /* Medium devices (landscape tablets, 768px and up) */ @media only screen and (min-width: 768px) { } /* Large devices (laptops/desktops, 992px and up) */ @media only screen and (min-width: 992px) { } /* Extra large devices (large laptops and desktops, 1200px and up) */ @media only screen and (min-width: 1200px) { } /* Advanced Media Queries */ @media (max-width: 768px) and (orientation: portrait) { } @media (prefers-color-scheme: dark) { } @media (hover: hover) { }

7. Essential Responsive Properties

/* Box Model */ * { box-sizing: border-box; /* Padding/border ko width mein include kare */ } /* Responsive Images */ img { max-width: 100%; /* Container se bada nahi hoga */ height: auto; /* Aspect ratio maintain kare */ } /* Responsive Typography */ html { font-size: 16px; /* Root font size */ } .responsive-heading { font-size: clamp(1.5rem, 4vw, 3rem); /* Fluid typography */ } /* Container Queries (New!) */ @container (min-width: 400px) { .card { display: grid; grid-template-columns: 1fr 2fr; } } /* Flexbox for Equal Height */ .equal-height-container { display: flex; flex-wrap: wrap; } .equal-height-item { flex: 1 1 300px; /* Grow, shrink, basis */ } /* Grid for Complex Layouts */ .responsive-grid { display: grid; grid-template-columns: repeat(auto-fit, minmax(250px, 1fr)); gap: 2rem; }
Responsive Design Best Practices:
1. Mobile First: Chhote screen ke liye design karo, phir bade screen ke liye enhance karo
2. Flexible Units: rem, %, vw/vh use karo instead of fixed px
3. Flexible Images: max-width: 100% aur height: auto always use karo
4. Touch Targets: Mobile pe buttons kam se kam 44px ke hone chahiye
5. Performance: CSS minify karo aur unnecessary rules avoid karo
6. Testing: Different devices pe test karo ya browser dev tools use karo

8. Modern CSS Features

/* CSS Custom Properties (Variables) */ :root { --primary-color: #667eea; --secondary-color: #764ba2; --container-padding: clamp(1rem, 5vw, 3rem); } .element { color: var(--primary-color); padding: var(--container-padding); } /* CSS Functions */ .modern-layout { width: min(90%, 1200px); /* Minimum value */ height: max(50vh, 400px); /* Maximum value */ font-size: clamp(1rem, 2.5vw, 2rem); /* Min, preferred, max */ margin: calc(100vh - 80px); /* Mathematical calculations */ } /* Aspect Ratio */ .video-container { aspect-ratio: 16 / 9; /* Maintain aspect ratio */ } /* CSS Subgrid */ .parent-grid { display: grid; grid-template-columns: repeat(3, 1fr); } .child-grid { display: grid; grid-template-columns: subgrid; /* Parent ke grid inherit kare */ }
